//title-9 palindrome number (leetcode-9)
#include<stdio.h>
bool isPalindrome(int x) {
    long res=0;
    int temp = x;
    while(temp!=0 && temp>0)
    {
        int r = temp%10;
        res = res*10+r;
        temp = temp/10;
    }
    if(res==x)
    {
        return true;
    }
    else
    {
        return false;
    }
}
void main()
{
    isPalindrome();
}
